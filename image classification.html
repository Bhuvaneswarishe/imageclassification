import numpy as np
import cv2
import matplotlib.pyplot as plt
from tensorflow.keras.models import load_model

# Load the trained model
model = load_model("animal_classification_model.h5")

# Class indices from the training generator
class_indices = {'cats': 0, 'dogs': 1, 'snakes': 2}  # Replace with train_generator.class_indices
class_labels = {v: k for k, v in class_indices.items()}  # Reverse dictionary

def predict_image_with_label(image_path, model, class_labels):
   
    # Load the image using OpenCV
    image = cv2.imread(image_path)
    if image is None:
        print("Error: Unable to load the image. Please check the path.")
        return

    # Convert BGR to RGB (OpenCV uses BGR, but matplotlib uses RGB)
    image_rgb = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)

    # Resize the image to match the input size of the model
    resized_image = cv2.resize(image_rgb, (150, 150))

    # Normalize pixel avalues to [0, 1]
    normalized_image = resized_image / 255.0

    # Expand dimensions to match the input shape expected by the model
    input_image = np.expand_dims(normalized_image, axis=0)

    # Predict using the model
    predictions = model.predict(input_image)

    # Get the class index with the highest probability
    predicted_class_index = np.argmax(predictions)

    # Map predicted index to class label
    if predicted_class_index in class_labels:
        predicted_class_label = class_labels[predicted_class_index]
    else:
        predicted_class_label = "Unknown"

    # Display the image with the predicted label
    plt.figure(figsize=(6, 6))
    plt.imshow(image_rgb)  # Display the image
    plt.title(f"Prediction: {predicted_class_label}")  # Set the title with predicted label
    plt.axis('off')  # Turn off axis
    plt.show()  # Show the image with title
